listening() {
    if [ $# -eq 0 ]; then
        sudo lsof -iTCP -sTCP:LISTEN -n -P
    elif [ $# -eq 1 ]; then
        sudo lsof -iTCP -sTCP:LISTEN -n -P | grep -i --color $1
    else
        echo "Usage: listening [pattern]"
    fi
}

zsh_terraform() {
  # break if there is no .terraform directory

    local tf_workspace=$(terraform workspace show)
    echo -n "$tf_workspace"
  fi
}
